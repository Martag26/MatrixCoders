Variables y constantes

Template literals y operadores

// Nullish coalescing
const config = null ?? { debug: false };   // { debug: false }

// Optional chaining
const ciudad = usuario?.direccion?.ciudad ?? "desconocida";

Arrow functions

// Expresión
const suma = (a, b) => a + b;

// Bloque
const saludar = (nombre) => {
  const msg = `Hola, ${nombre}`;
  return msg.toUpperCase();
};

// Callback
const pares = [1,2,3,4,5].filter(n => n % 2 === 0);

Spread y Rest

// Rest: agrupa argumentos en array
function log(nivel, ...mensajes) {
  console.log(`[${nivel}]`, mensajes.join(" "));
}

// Spread: expande iterables
const a = [1, 2];
const b = [3, 4];
const todo = [...a, ...b];          // [1,2,3,4]

const defaults = { timeout: 5000 };
const cfg = { ...defaults, timeout: 3000, retries: 3 };
