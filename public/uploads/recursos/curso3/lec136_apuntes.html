Funciones de primera clase

// Asignadas a variables
const multiplicar = function(a, b) { return a * b; };

// Pasadas como argumentos (callback)
[1,2,3].forEach(function(n) { console.log(n); });

// Retornadas desde otras funciones (closure)
function contador(inicio = 0) {
  let n = inicio;
  return {
    incrementar: () => ++n,
    valor: () => n
  };
}
const c = contador(10);
c.incrementar();  // 11

Estructuras de control

// if / else if / else
if (nota >= 9) { console.log("Sobresaliente"); }
else if (nota >= 5) { console.log("Aprobado"); }
else { console.log("Suspenso"); }

// switch con break
switch (metodo) {
  case "GET":  manejarGet(); break;
  case "POST": manejarPost(); break;
  default:     res.status(405).send("No permitido");
}

// for...of (iterable)
for (const ruta of rutas) { router.use(ruta); }

// for...in (claves de objeto)
for (const clave in config) { console.log(clave, config[clave]); }

Callbacks, Promises y async/await

// Callback (patrón Node clásico: error-first)
fs.readFile("datos.json", "utf8", (err, data) => {
  if (err) throw err;
  console.log(JSON.parse(data));
});

// Promise
fetch("https://api.ejemplo.com/usuarios")
  .then(res => res.json())
  .then(data => console.log(data))
  .catch(err => console.error(err));

// async/await (recomendado)
async function obtenerUsuarios() {
  try {
    const res  = await fetch("/api/usuarios");
    const data = await res.json();
    return data;
  } catch (err) {
    console.error("Error:", err.message);
  }
}
