Estructura del proyecto Node.js + Express

matrixCoders · Apuntes de lección · Backend con Node.js y Bases de Datos

Arquitectura recomendada

mi-api/
├── src/
│   ├── config/          # variables de entorno, db, logger
│   ├── controllers/     # lógica de cada recurso
│   ├── middlewares/     # auth, validación, errores
│   ├── models/          # esquemas Mongoose / clases BD
│   ├── routes/          # definición de rutas Express
│   ├── services/        # lógica de negocio
│   └── app.js           # configuración Express
├── tests/
├── .env
├── .env.example
├── package.json
└── server.js            # punto de entrada

Configuración de Express

// src/app.js
const express = require("express");
const helmet  = require("helmet");
const cors    = require("cors");

const app = express();

app.use(helmet());                          // cabeceras de seguridad
app.use(cors({ origin: process.env.CORS_ORIGIN }));
app.use(express.json({ limit: "10kb" }));  // parse body JSON

// Rutas
app.use("/api/v1/usuarios",  require("./routes/usuarios"));
app.use("/api/v1/productos", require("./routes/productos"));

// Health-check
app.get("/health", (req, res) => res.json({ status: "ok" }));

module.exports = app;
