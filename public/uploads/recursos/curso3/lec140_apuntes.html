Router y controlador

// routes/usuarios.js
const router = require("express").Router();
const ctrl   = require("../controllers/usuariosCtrl");
const auth   = require("../middlewares/auth");

router.get   ("/",    auth,        ctrl.listar);
router.get   ("/:id", auth,        ctrl.obtener);
router.post  ("/",                  ctrl.crear);
router.put   ("/:id", auth,        ctrl.actualizar);
router.delete("/:id", auth,        ctrl.eliminar);

module.exports = router;

// controllers/usuariosCtrl.js
const UsuarioService = require("../services/UsuarioService");

exports.listar = async (req, res, next) => {
  try {
    const { pagina = 1, limite = 20 } = req.query;
    const data = await UsuarioService.listar({ pagina, limite });
    res.json(data);
  } catch (err) { next(err); }
};

Servicio y acceso a datos (SQLite / Knex)

// services/UsuarioService.js
const db = require("../config/db");

exports.listar = async ({ pagina, limite }) => {
  const offset = (pagina - 1) * limite;
  const rows   = await db("usuarios")
    .select("id", "nombre", "email", "creado_en")
    .limit(limite).offset(offset);
  const [{ total }] = await db("usuarios").count("* as total");
  return { data: rows, total, pagina, limite };
};

exports.crear = async (datos) => {
  const [id] = await db("usuarios").insert(datos);
  return db("usuarios").where({ id }).first();
};

Validación con Joi

const Joi = require("joi");

const esquemaUsuario = Joi.object({
  nombre:   Joi.string().min(2).max(100).required(),
  email:    Joi.string().email().required(),
  password: Joi.string().min(8).pattern(/[A-Z]/).required(),
  rol:      Joi.string().valid("admin", "user").default("user"),
});

// Middleware de validación
function validar(esquema) {
  return (req, res, next) => {
    const { error, value } = esquema.validate(req.body, { abortEarly: false });
    if (error) {
      const errores = error.details.map(d => d.message);
      return res.status(400).json({ errores });
    }
    req.body = value;
    next();
  };
}
