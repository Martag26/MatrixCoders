Declaración de variables

const PI    = 3.1416;     // inmutable (referencia)
let   total = 0;           // mutable, scope de bloque
var   legado = "evitar";   // scope de función — usar sólo en código antiguo

Regla práctica: const por defecto, let si necesitas reasignar, var casi nunca.

const a = [1,2];
const b = [...a, 3, 4];     // spread → [1,2,3,4]

function suma(...nums) {     // rest → array
  return nums.reduce((acc, n) => acc + n, 0);
}

Estructuras de control

// if / else
if (edad >= 18) { acceso(); }
else            { rechazar(); }

// switch
switch (rol) {
  case "admin":    panelAdmin(); break;
  case "editor":   panelEditor(); break;
  default:         panelInvitado();
}

// for...of (valores)
for (const item of carrito) { console.log(item); }

// for...in (claves)
for (const k in user) { console.log(k, user[k]); }

// while
while (intentos < 3) { intentos++; }
