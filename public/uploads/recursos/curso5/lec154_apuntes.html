Funciones, closures y estructuras de control

matrixCoders · Apuntes de lección · JavaScript

Definición de funciones

// Declaración (se "hoistea")
function saludar(nombre) {
  return `Hola, ${nombre}`;
}

// Expresión
const saludar2 = function(nombre) { return `Hola, ${nombre}`; };

// Arrow (no tiene su propio this)
const saludar3 = nombre => `Hola, ${nombre}`;
const suma     = (a, b) => a + b;
const obj      = (x, y) => ({ x, y });   // paréntesis para devolver objeto

Parámetros: defaults, rest, destructuring

function crearUsuario({ nombre, edad = 18, rol = "user" } = {}) {
  return { nombre, edad, rol };
}

crearUsuario({ nombre: "Ana" });             // { nombre:"Ana", edad:18, rol:"user" }

function logAll(prefijo, ...mensajes) {
  mensajes.forEach(m => console.log(prefijo, m));
}
logAll("[INFO]", "uno", "dos", "tres");

Funciones de orden superior y callbacks

// Una función que recibe otra
function repetir(n, fn) {
  for (let i = 0; i < n; i++) fn(i);
}
repetir(3, i => console.log("iteración", i));

// Una función que devuelve otra (closure)
function multiplicador(factor) {
  return n => n * factor;
}
const x2 = multiplicador(2);
x2(10);   // 20

Closures

Un closure es una función que recuerda las variables del scope donde fue creada.

function contador() {
  let n = 0;
  return {
    incrementar: () => ++n,
    valor:       () => n,
    reset:       () => (n = 0),
  };
}
const c = contador();
c.incrementar(); c.incrementar();
c.valor();   // 2 — n vive aunque contador() ya terminó

Manejo del this

const obj = {
  nombre: "Ana",
  saludar() { console.log(this.nombre); },        // this → obj
  flecha:   () => console.log(this.nombre),       // this → window/undefined
};
obj.saludar();   // "Ana"
obj.flecha();    // undefined

// bind / call / apply
function decir(saludo) { return `${saludo}, ${this.nombre}`; }
decir.call({ nombre: "Ana" }, "Hola");   // "Hola, Ana"
