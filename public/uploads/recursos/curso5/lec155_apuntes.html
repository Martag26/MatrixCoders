Clases ES6

class Persona {
  #id;                        // campo privado
  static contador = 0;        // estático

  constructor(nombre, edad) {
    this.nombre = nombre;
    this.edad   = edad;
    this.#id    = ++Persona.contador;
  }

  get id() { return this.#id; }
  saludar() { return `Hola, soy ${this.nombre}`; }

  static fromString(str) {
    const [nombre, edad] = str.split(",");
    return new Persona(nombre, parseInt(edad));
  }
}

const ana = new Persona("Ana", 28);
ana.id;                          // 1
Persona.fromString("Luis,30");   // instancia desde texto

Herencia

class Empleado extends Persona {
  constructor(nombre, edad, salario) {
    super(nombre, edad);          // obligatorio antes de this
    this.salario = salario;
  }

  saludar() {
    return super.saludar() + ` y gano ${this.salario}€`;
  }
}

const e = new Empleado("Luis", 30, 35000);
e instanceof Empleado;  // true
e instanceof Persona;   // true

Módulos ES (import / export)

// math.js
export const PI = 3.1416;
export function sumar(a, b) { return a + b; }
export default function multiplicar(a, b) { return a * b; }

// app.js
import multiplicar, { PI, sumar } from "./math.js";
import * as math from "./math.js";   // namespace

console.log(sumar(2,3), multiplicar(4,5), PI);

// Carga dinámica (devuelve Promise)
const m = await import("./math.js");
m.sumar(1, 2);

Para usarlos en el navegador: <script type="module" src="app.js"></script>.

Prototipos (lo que hay debajo)

Las clases JS son azúcar sintáctico sobre los prototipos. Cada objeto tiene una referencia interna a otro objeto (su prototipo) del que hereda propiedades.

const a = { saludar() { return "Hola"; } };
const b = Object.create(a);
b.saludar();        // "Hola" — encontrado en el prototipo
Object.getPrototypeOf(b) === a;   // true
