try / catch / finally

function dividir(a, b) {
  if (b === 0) throw new RangeError("División por cero");
  return a / b;
}

try {
  const r = dividir(10, 0);
} catch (err) {
  if (err instanceof RangeError) {
    console.error("Error matemático:", err.message);
  } else {
    throw err;   // re-lanza lo desconocido
  }
} finally {
  console.log("Siempre se ejecuta");
}

Errores personalizados

class ApiError extends Error {
  constructor(message, status = 500) {
    super(message);
    this.name   = "ApiError";
    this.status = status;
  }
}

throw new ApiError("Usuario no encontrado", 404);

Promesas

// Crear una promesa
const cargar = new Promise((resolve, reject) => {
  setTimeout(() => Math.random() > 0.5
    ? resolve("OK")
    : reject(new Error("falló")), 500);
});

cargar
  .then(res  => console.log("éxito:", res))
  .catch(err => console.error("error:", err.message))
  .finally(() => console.log("terminó"));

// Combinadores útiles
Promise.all([p1, p2]);          // todas o ninguna
Promise.allSettled([p1, p2]);   // siempre resuelve, con estados
Promise.race([p1, p2]);         // la primera que termine
Promise.any([p1, p2]);          // la primera que cumpla

async / await

Permite escribir código asíncrono con apariencia síncrona.

async function obtenerUsuarios() {
  try {
    const res = await fetch("/api/usuarios");
    if (!res.ok) throw new ApiError("HTTP " + res.status, res.status);
    const data = await res.json();
    return data;
  } catch (err) {
    console.error(err);
    return [];
  }
}

// Paralelo (no esperes secuencialmente sin necesidad)
const [users, posts] = await Promise.all([
  fetch("/api/users").then(r => r.json()),
  fetch("/api/posts").then(r => r.json()),
]);

Captura global de errores

// Errores síncronos no manejados
window.addEventListener("error", e => {
  console.log("global:", e.message, e.filename, e.lineno);
});

// Promesas rechazadas sin .catch
window.addEventListener("unhandledrejection", e => {
  console.log("promise rechazada:", e.reason);
});
