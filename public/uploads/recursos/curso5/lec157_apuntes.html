Modificar el DOM

// Texto y HTML
el.textContent = "Hola";
el.innerHTML   = "<strong>Hola</strong>";   // peligroso si viene de usuario

// Atributos y clases
el.setAttribute("data-id", "42");
el.dataset.id;            // "42"
el.classList.add("activa");
el.classList.toggle("oculto");
el.classList.contains("activa");

// Estilos
el.style.color = "tomato";

// Crear e insertar
const li = document.createElement("li");
li.textContent = "Nuevo";
lista.appendChild(li);
lista.insertAdjacentHTML("beforeend", "<li>X</li>");
li.remove();

Eventos

boton.addEventListener("click", e => {
  e.preventDefault();        // cancela acción nativa
  e.stopPropagation();       // detiene burbuja
  console.log("clic en", e.target);
});

// Tipos comunes
input.addEventListener("input",  e => console.log(e.target.value));
input.addEventListener("change", e => save(e.target.value));
form.addEventListener("submit",  e => { e.preventDefault(); enviar(); });
window.addEventListener("keydown", e => { if (e.key === "Escape") cerrar(); });

Delegación de eventos

En vez de un listener por elemento, escucha en el padre:

document.querySelector("#lista").addEventListener("click", e => {
  const item = e.target.closest("li.eliminable");
  if (!item) return;
  item.remove();
});

Es más eficiente y funciona para elementos añadidos dinámicamente.

Fetch — peticiones HTTP

// GET
const res  = await fetch("/api/posts");
const data = await res.json();

// POST con JSON
await fetch("/api/login", {
  method:  "POST",
  headers: { "Content-Type": "application/json" },
  body:    JSON.stringify({ email, password }),
});

// Manejo correcto de errores
const r = await fetch(url);
if (!r.ok) throw new Error(`HTTP ${r.status}`);

// Cancelar petición tras 5s
const ctrl = new AbortController();
setTimeout(() => ctrl.abort(), 5000);
await fetch(url, { signal: ctrl.signal });
