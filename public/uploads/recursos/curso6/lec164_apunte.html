Clases y objetos

<?php
class Producto {
    // Propiedades
    public string $nombre;
    private float $precio;

    // Constructor
    public function __construct(string $nombre, float $precio) {
        $this->nombre = $nombre;
        $this->precio = $precio;
    }

    // Getter
    public function getPrecio(): float {
        return $this->precio;
    }

    // Método
    public function aplicarDescuento(float $pct): void {
        $this->precio *= (1 - $pct / 100);
    }
}

// Crear objeto
$prod = new Producto("Laptop", 999.00);
$prod->aplicarDescuento(10);
echo $prod->getPrecio(); // 899.1
?>

Herencia

<?php
class Animal {
    public function respirar() { echo "Respiro..."; }
}

class Perro extends Animal {
    public function ladrar() { echo "¡Guau!"; }
}

$p = new Perro();
$p->respirar(); // heredado
$p->ladrar();
?>

Interfaces y clases abstractas

<?php
interface Pagable {
    public function pagar(float $monto): bool;
}

class TarjetaCredito implements Pagable {
    public function pagar(float $monto): bool {
        echo "Pagando $$monto con tarjeta.";
        return true;
    }
}
?>
