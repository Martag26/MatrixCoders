Operaciones CRUD

CRUD = Create, Read, Update, Delete. Son las cuatro operaciones básicas de cualquier aplicación que persiste datos.

CREATE — Insertar registro

<?php
function crearUsuario(PDO $db, string $nombre, string $email): int {
    $stmt = $db->prepare("INSERT INTO users (nombre, email) VALUES (?, ?)");
    $stmt->execute([$nombre, $email]);
    return (int) $db->lastInsertId(); // ID del nuevo registro
}
?>

READ — Leer registros

<?php
// Todos los usuarios
$usuarios = $db->query("SELECT * FROM users ORDER BY nombre")->fetchAll();

// Un usuario por ID
$stmt = $db->prepare("SELECT * FROM users WHERE id = ?");
$stmt->execute([$id]);
$usuario = $stmt->fetch(); // false si no existe

// Mostrar en HTML
foreach ($usuarios as $u) {
    echo "<li>{$u['nombre']} — {$u['email']}</li>";
}
?>

UPDATE — Actualizar registro

<?php
function actualizarEmail(PDO $db, int $id, string $nuevoEmail): int {
    $stmt = $db->prepare("UPDATE users SET email = ? WHERE id = ?");
    $stmt->execute([$nuevoEmail, $id]);
    return $stmt->rowCount(); // filas afectadas
}
?>

DELETE — Eliminar registro

<?php
function eliminarUsuario(PDO $db, int $id): bool {
    $stmt = $db->prepare("DELETE FROM users WHERE id = ?");
    $stmt->execute([$id]);
    return $stmt->rowCount() > 0;
}
?>

Transacciones

<?php
$db->beginTransaction();
try {
    $db->exec("UPDATE cuentas SET saldo = saldo - 100 WHERE id = 1");
    $db->exec("UPDATE cuentas SET saldo = saldo + 100 WHERE id = 2");
    $db->commit(); // Ambas o ninguna
} catch (Exception $e) {
    $db->rollBack(); // Revertir todo
    throw $e;
}
?>
💡 Usa transacciones siempre que necesites que varias operaciones se completen juntas o ninguna.
