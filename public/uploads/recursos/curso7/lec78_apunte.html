La clase Math II: Números aleatorios y redondeos

Números aleatorios

// Math.random() → double entre 0.0 (incluido) y 1.0 (excluido)
double r = Math.random();

// Entero aleatorio entre min y max (incluidos)
int min = 1, max = 6;
int dado = (int) (Math.random() * (max - min + 1)) + min;

// Usando Random (más flexible)
import java.util.Random;
Random rnd = new Random();
int num  = rnd.nextInt(100);     // 0-99
int num2 = rnd.nextInt(6) + 1;  // 1-6

Redondeos

Math.round(3.5)     // 4   (redondeo estándar)
Math.floor(3.9)     // 3.0 (siempre hacia abajo)
Math.ceil(3.1)      // 4.0 (siempre hacia arriba)

// Formato con 2 decimales
System.out.printf("%.2f%n", 3.14159); // "3.14"
String s = String.format("%.2f", 3.14159); // "3.14"

Ejemplo: simulación de dados

import java.util.Random;

public class SimuladorDado {
    public static void main(String[] args) {
        Random r = new Random();
        int dado1 = r.nextInt(6) + 1;
        int dado2 = r.nextInt(6) + 1;
        System.out.println("Dado 1: " + dado1);
        System.out.println("Dado 2: " + dado2);
        System.out.println("Total: " + (dado1 + dado2));
    }
}
