Ejemplos de uso de matrices simples

Suma de filas y columnas

int[][] tabla = {
    {1, 2, 3},
    {4, 5, 6},
    {7, 8, 9}
};

// Suma por filas
for (int i = 0; i < tabla.length; i++) {
    int sumaFila = 0;
    for (int j = 0; j < tabla[i].length; j++) {
        sumaFila += tabla[i][j];
    }
    System.out.println("Fila " + i + ": " + sumaFila);
}
// Fila 0: 6, Fila 1: 15, Fila 2: 24

Buscar el máximo

int[] valores = {34, 17, 89, 5, 62};
int max = valores[0];
int idxMax = 0;

for (int i = 1; i < valores.length; i++) {
    if (valores[i] > max) {
        max    = valores[i];
        idxMax = i;
    }
}
System.out.println("Máximo: " + max + " en índice " + idxMax);

Rotar array (desplazamiento)

int[] arr = {1, 2, 3, 4, 5};
int primero = arr[0];

for (int i = 0; i < arr.length - 1; i++) {
    arr[i] = arr[i + 1];
}
arr[arr.length - 1] = primero;
// [2, 3, 4, 5, 1]
