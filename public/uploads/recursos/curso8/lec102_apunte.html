Creación de componentes funcionales

Formas de escribir un componente funcional

// 1. Función declarada
function Boton() {
  return <button>Clic aquí</button>;
}

// 2. Arrow function (más común en proyectos modernos)
const Boton = () => {
  return <button>Clic aquí</button>;
};

// 3. Return implícito (para JSX simple)
const Boton = () => <button>Clic aquí</button>;

Componente con lógica interna

const Tarjeta = () => {
  const fecha = new Date().toLocaleDateString('es-ES');
  const esDiaPar = new Date().getDate() % 2 === 0;

  return (
    <div className="tarjeta">
      <h2>Hoy es {fecha}</h2>
      {esDiaPar && <p>¡Día par!</p>}
    </div>
  );
};

Renderizado condicional

// 1. Ternario
{activo ? <Online /> : <Offline />}

// 2. Short-circuit (&&) — solo renderiza si condición es true
{mensajes.length > 0 && <Badge count={mensajes.length} />}

// 3. null = no renderiza nada
if (!visible) return null;
