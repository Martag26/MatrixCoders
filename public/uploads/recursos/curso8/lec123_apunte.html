React

Proyecto: Buscador de información en tiempo real

Objetivo

Crear un buscador de usuarios de GitHub que muestre el avatar, nombre y bio de cualquier usuario al escribir en un input.

Estructura del componente

import { useState, useEffect } from 'react';

function BuscadorGitHub() {
  const [query, setQuery]     = useState('');
  const [usuario, setUsuario] = useState(null);
  const [cargando, setCargando] = useState(false);
  const [error, setError]     = useState('');

  useEffect(() => {
    if (!query) { setUsuario(null); return; }

    const timer = setTimeout(() => {
      setCargando(true);
      setError('');
      fetch(`https://api.github.com/users/${query}`)
        .then(r => { if (!r.ok) throw new Error('Usuario no encontrado'); return r.json(); })
        .then(data => setUsuario(data))
        .catch(e => setError(e.message))
        .finally(() => setCargando(false));
    }, 500);  // debounce: espera 500ms antes de buscar

    return () => clearTimeout(timer);
  }, [query]);

  return (
    <div>
      <input
        value={query}
        onChange={e => setQuery(e.target.value)}
        placeholder="Buscar usuario GitHub..."
      />
      {cargando && <p>Buscando...</p>}
      {error    && <p style={{color:'red'}}>{error}</p>}
      {usuario  && (
        <div>
          <img src={usuario.avatar_url} width="80" alt="avatar" />
          <h2>{usuario.name}</h2>
          <p>{usuario.bio}</p>
          <p>⭐ {usuario.public_repos} repos</p>
        </div>
      )}
    </div>
  );
}
💡 El patrón debounce (setTimeout + clearTimeout en el cleanup) evita hacer una petición por cada tecla. Espera a que el usuario deje de escribir.
