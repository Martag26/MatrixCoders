Node.js

Desarrollo de API desde cero con Express

¿Qué es Express?

Express es el framework web más popular para Node.js. Minimalista y flexible, proporciona un sistema de routing y middleware para construir APIs y aplicaciones web.

Instalación y setup básico

# Crear proyecto
mkdir mi-api && cd mi-api
npm init -y
npm install express
npm install --save-dev nodemon
// index.js
const express = require('express');
const app     = express();

app.use(express.json());  // parsear body JSON

app.get('/', (req, res) => {
  res.json({ mensaje: 'API funcionando' });
});

app.listen(3000, () => console.log('Servidor en :3000'));

Verbos HTTP y rutas

const productos = [
  { id: 1, nombre: 'Teclado', precio: 49 },
  { id: 2, nombre: 'Ratón',   precio: 29 },
];

// GET — obtener todos
app.get('/api/productos', (req, res) => {
  res.json(productos);
});

// GET — obtener por ID
app.get('/api/productos/:id', (req, res) => {
  const prod = productos.find(p => p.id === +req.params.id);
  if (!prod) return res.status(404).json({ error: 'No encontrado' });
  res.json(prod);
});

// POST — crear
app.post('/api/productos', (req, res) => {
  const nuevo = { id: Date.now(), ...req.body };
  productos.push(nuevo);
  res.status(201).json(nuevo);
});

// PUT — actualizar
app.put('/api/productos/:id', (req, res) => {
  const idx = productos.findIndex(p => p.id === +req.params.id);
  if (idx === -1) return res.status(404).json({ error: 'No encontrado' });
  productos[idx] = { ...productos[idx], ...req.body };
  res.json(productos[idx]);
});

// DELETE — eliminar
app.delete('/api/productos/:id', (req, res) => {
  const idx = productos.findIndex(p => p.id === +req.params.id);
  if (idx === -1) return res.status(404).json({ error: 'No encontrado' });
  productos.splice(idx, 1);
  res.status(204).send();
});
