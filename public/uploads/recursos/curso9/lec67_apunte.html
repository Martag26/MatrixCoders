MySQL con Node.js y Buenas Prácticas

Conectar Node.js con MySQL

npm install mysql2
// db/connection.js
const mysql = require('mysql2/promise');

const pool = mysql.createPool({
  host:     process.env.DB_HOST || 'localhost',
  user:     process.env.DB_USER || 'root',
  password: process.env.DB_PASS || '',
  database: process.env.DB_NAME || 'mi_bd',
  waitForConnections: true,
  connectionLimit: 10,
});

module.exports = pool;

CRUD con async/await y prepared statements

const pool = require('../db/connection');

// Obtener todos
const getAll = async (req, res) => {
  try {
    const [rows] = await pool.query('SELECT * FROM productos');
    res.json(rows);
  } catch (e) {
    res.status(500).json({ error: e.message });
  }
};

// Crear con prepared statement (evita SQL injection)
const create = async (req, res) => {
  const { nombre, precio } = req.body;
  try {
    const [result] = await pool.execute(
      'INSERT INTO productos (nombre, precio) VALUES (?, ?)',
      [nombre, precio]
    );
    res.status(201).json({ id: result.insertId, nombre, precio });
  } catch (e) {
    res.status(500).json({ error: e.message });
  }
};

Validación de datos de entrada

npm install joi  // o express-validator
const Joi = require('joi');

const schema = Joi.object({
  nombre: Joi.string().min(2).max(100).required(),
  precio: Joi.number().positive().required(),
});

const { error, value } = schema.validate(req.body);
if (error) return res.status(400).json({ error: error.details[0].message });

Manejo global de errores

// Middleware de error (4 parámetros)
app.use((err, req, res, next) => {
  console.error(err.stack);
  res.status(err.status || 500).json({
    error: process.env.NODE_ENV === 'production'
      ? 'Error interno'
      : err.message
  });
});
⚠️ Usa siempre pool.execute(query, [params]) con parámetros para evitar inyección SQL. Nunca concatenes datos del usuario en una query.
